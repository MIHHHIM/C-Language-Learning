### Test-02.c

> 该程序用于打印字符串 整型 字符型和浮点型四种不同类型的值

```
// 包含标准输入输出头文件 提供printf等输入输出函数的声明
#include <stdio.h>

// 定义主函数 作为程序执行的入口点 返回值类型为整数
int main() 
{
    // 定义一个整型变量n 并初始化为100
    int n = 100;

    // 使用printf函数输出字符串%s是字符串的格式控制符
    printf("%s\n", "Hello World");
    
    // 输出整数n的值%d是带符号十进制整数的格式控制符
    printf("%d\n", n);
    
    // 输出字符'a'%c是字符型的格式控制符
    printf("%c\n", 'a');
    
    // 输出浮点数3.14%f是浮点型的格式控制符
    printf("%f\n", 3.14);

    // 主函数返回0 表示程序正常结束
    return 0;
}
```

### Test-03.c

> 该程序演示了 ASCII 码的概念 通过整数或字符运算来打印特定的字符

```
// 包含标准输入输出头文件 用于使用printf函数
#include <stdio.h>

// 主函数 程序执行的入口
int main() 
{
    // 使用%c格式控制符输出ASCII码值为88的字符 即大写字母'X'
    printf("%c\n", 88);
    
    // 直接输出字符'x'
    printf("%c\n", 'x');
    
    // 字符'x'的ASCII码是120 减去32后得到88 对应大写字母'X'
    // 这体现了小写字母与大写字母的ASCII码值通常相差32的特性
    printf("%c\n", 'x' - 32);

    // 主函数返回0 表示程序正常结束
    return 0;
}
```

### Test-04.c

> 该程序通过循环打印出所有可见的 ASCII 字符(码值从 32 到 127)

```
// 包含标准输入输出头文件 用于使用printf函数
#include <stdio.h>

// 主函数 程序执行的入口
int main() 
{
    // 使用for循环 变量i从32开始 到127结束
    // 这个范围 (32-127) 包含了所有可打印的ASCII字符0-31是控制字符 通常不可见
    for (int i = 32; i <= 127; i++) 
    {
        // 以字符形式输出当前i对应的ASCII字符 后面加一个空格用于分隔
        printf("%c ", i);
        
        // 当i是16的倍数减1时(即每输出16个字符后) 输出一个换行符
        // 这使得输出结果更加整齐 每行显示16个字符
        if (i % 16 == 15) 
        {
            printf("\n");
        }
    }
    // 主函数返回0 表示程序正常结束
    return 0;
}
```

### Test-05.c

> 该程序演示了转义字符的用法 用于打印一些特殊符号 如单引号 双引号 反斜杠等

```
// 包含标准输入输出头文件 用于使用printf函数
#include <stdio.h>

// 主函数 程序执行的入口点
int main() 
{
    // 使用转义序列\'来表示单引号字符本身
    printf("%c\n", '\'');
    
    // 使用转义序列\"来表示双引号字符本身
    printf("%s\n", "\"");
    
    // 在字符串中 使用转义序列\\来表示一个反斜杠 以正确输出文件路径
    printf("C:\\Test\\Code\\Hello.c\n");
    
    // \a是响铃转义序列 执行时会触发系统提示音(如果终端支持)
    printf("\a");
    
    // \130是一个八进制转义序列 表示的ASCII码值为 (1 * 8^2) + (3 * 8^1) + (0 * 8^0) = 64 + 24 = 88 对应字符'X'
    printf("%c\n", '\130');
    
    // \x30是一个十六进制转义序列 表示的ASCII码值为 (3 * 16^1) + (0 * 16^0) = 48 对应字符'0'
    printf("%c\n", '\x30');
    
    // 主函数返回0 表示程序正常结束
    return 0;
}
```

### Test-06.c

> 该程序用于计算并输出一个简单字符串的长度


### Test-07.c

> 该程序计算一个包含转义字符的字符串的长度

```
// 包含标准输入输出头文件 用于printf函数
#include <stdio.h>
// 包含字符串处理头文件 用于使用strlen函数
#include <string.h>

// 主函数 程序执行的入口点
int main() 
{
    // 计算字符串"C:\\Code\\Test\\555\\C"的长度
    // 在C语言字符串中 \\是一个转义序列 表示一个'\'字符
    // 因此 该字符串的实际内容是 "C:\Code\Test\555\C"
    // 长度计算为: C(1) + :(1) + \(1) + Code(4) + \(1) + Test(4) + \(1) + 555(3) + \(1) + C(1) = 18
    int len = strlen("C:\\Code\\Test\\555\\C"); 
    
    // 输出计算出的长度 结果为18
    printf("%d\n", len);
    
    // 主函数返回0 表示程序正常结束
    return 0;
}
```

### Test-08.c

> 该程序是一个布尔条件判断示例 条件为 `true`

```
// 包含标准输入输出库 用于使用printf函数
#include <stdio.h>
// 包含布尔类型头文件 用于使用bool类型和true/false常量
#include <stdbool.h>

// 程序主函数 程序从这里开始执行
int main() 
{
    // 定义布尔类型变量a 并初始化为true (真)
    bool a = true;

    // if语句判断变量a的值因为a为true 所以条件成立
    if (a) 
    {
        // 条件为真 执行花括号内的语句 打印"Hello World"
        printf("Hello World\n");
    }

    // 主函数返回0 表示程序正常结束
    return 0;
}
```

### Test-10.c

> 该程序是另一个布尔条件判断的示例 当条件为 `true` 时 打印一个整型变量的值

```
// 引入标准输入输出头文件 用于使用 printf 函数
#include <stdio.h>
// 引入布尔类型头文件 用于使用 bool 类型和 true/false 常量
#include <stdbool.h>

// 程序的主入口函数
int main() 
{
    // 定义一个整型变量 a 并赋值为 100
    int a = 100;
    // 定义一个布尔型变量 b 并赋值为 true (真)
    bool b = true;

    // 条件判断语句 检查变量 b 的值是否为真
    if (b) 
    {
        // 由于 b 为 true 条件成立 执行此处的代码
        // 使用 %d 格式控制符 打印整型变量 a 的值 (100) 并换行
        printf("%d\n", a);
    }

    // 主函数返回0 表示程序正常结束
    return 0;
}
```
